#Create lists to hold relevant data that will be taken from the usatoday data source
playerList = []
dateList = []
teamList = []
chargeList = []

#Extracting names of arrested players and appending to list
for player in playerArrestsSoup.find_all("td", {'class':'left arrest-name'}):
    playerList.append(player.text)
#Extracting date of arrest and appending to list
for date in playerArrestsSoup.find_all("td", {'class':'left arrest-date'}):
    dateList.append(date.text)
#Extracting team that arrested player plays for and appending to list     
for team in playerArrestsSoup.find_all("td", {'class':'left arrest-narrow'}):
    teamList.append(team.text)
#Extracting crime that arrested player was charged for and appending to list    
for charge in playerArrestsSoup.find_all("td", {'class':'left arrest-midsize'}):
    chargeList.append(charge.text)

#Here we will look into the types of crimes being committed by NFL Players

#First create a dictionary where the key will be the type of crime (DUI,Drugs,etc)
#and the value will be the number of players charged with that type of crime
chargeDict = {}
#Iterate through the arrestDict from above
for player in arrestDict:
    #if a type of crime (index 2)is not in the new chargeDict, then it is assigned as a key
    if arrestDict[player][2] not in chargeDict:
        chargeDict[arrestDict[player][2]] = 1
    #otherwise, the value (number of players charged with that crime) will be increased by 1    
    else:
        chargeDict[arrestDict[player][2]] += 1

#Because the list contains many, many unique crimes committed by NFL players, I decided to trim the list to exclude unique crimes with single occurences
#I use .items() method to return the charge, count key value pairs, and delete the pair if 
#the count (occurence of that crime) is unique (<1)
for charge, count in dict(chargeDict).items():
    if count < 2:
        del chargeDict[charge]
#finally, using zip, I create charge and count variables to hold the keys and values from the chargeDict
charge, count = zip(*chargeDict.items())

#This function calculates the win% of a team before one of their players was charged with a crime, and the win% for the rest of the season
#The function returns the change in win percentage after a player was charged with a crime:
#I wanted to see if there was any clear indication if player being charged with a crime affects team win %


#The argument passed into the function is the name of the player that was charged
def teamPerformanceArrests(Player):
    #here I create some variables that I need to calculate win%
    winsb4Arrest = 0
    winsAfterArrest = 0
    lossesB4Arrest = 0
    lossesAfterArrest = 0
    #open the file and designate a header
    NFLGameLog = open('NFL Game Logs cleaned.csv',"r", encoding='utf8')
    header = NFLGameLog.readline().rstrip("\n")
    #I have a static date (month-day) to indicate the start of the season
    startOfSeason = "-09-05"
    #same thing for the end of the season
    endOfSeason = "-12-31"
    #check to see if the argument supplied is indeed a player that was charged
    if Player in arrestDict:
        #create variables to hold date of charge and team of player
        dateOfArrest = arrestDict[Player][0]
        playerTeam = arrestDict[Player][1]
        #print(dateOfArrest)
        #create a variable that takes the year of the crime and creates the date representing the start of that season
        adjustedStartofSeason = str(dateOfArrest[0:4])+str(startOfSeason)
        #turn date string into datetime object
        timeAdjustedStartofSeason = datetime.strptime(adjustedStartofSeason , '%Y-%m-%d').date()
        #create a variable that takes the year of the crime and creates the date representing the end of that season
        adjustedEndofSeason = str(dateOfArrest[0:4])+str(endOfSeason)
        #turn date string into datetime object
        TimeAdjustedEndofSeason = datetime.strptime(adjustedEndofSeason , '%Y-%m-%d').date()
        #create date range for season player was charged
        seasonOfArrest = pd.date_range(timeAdjustedStartofSeason, TimeAdjustedEndofSeason)
        #date range representing NFL season up to the crime
        seasonUpToArrest = pd.date_range(timeAdjustedStartofSeason, dateOfArrest)
        #date range representing NFL season after the crime
        seasonAfterArrest = pd.date_range(dateOfArrest, TimeAdjustedEndofSeason)
        #check if date player was charged falls within the NFL season
        #because most NFL players are charged in the offseason, this if statement immediately filters out most of the crime records
        if dateOfArrest in seasonOfArrest:
            #iterate through gamelog csv and split on commas
            for row in NFLGameLog:
                row = row.split(",")
                #if a game date falls within the season a player was charged with a crime
                if row[2] in seasonOfArrest:
                    #if the player's team played in that game
                    if playerTeam in row[5] or playerTeam in row[6]:
                        #if the players team won the game
                        if playerTeam in row[14]:
                            #if the game won was before the crime
                            if row[2] in seasonUpToArrest:
                                winsb4Arrest += 1
                            #if the game won was after the crime
                            if row[2] in seasonAfterArrest:
                                winsAfterArrest += 1
                        #if the team lost the game
                        if playerTeam not in row[14]:
                            #if the loss occured before the crime
                            if row[2] in seasonUpToArrest:
                                lossesB4Arrest += 1
                            #if the loss occured after the crime
                            if row[2] in seasonAfterArrest:
                                lossesAfterArrest += 1
            # if the loss values are 0, calculation cannot divide by 0, so filter these out 
            if lossesB4Arrest > 0 and lossesAfterArrest > 0:
                
                gamesb4Arrest = winsb4Arrest + lossesB4Arrest
                #calc win percentage before the crime
                winPercentageB4Arrest = winsb4Arrest / gamesb4Arrest
                gamesAfterArrest = winsAfterArrest + lossesAfterArrest
                #calculate win% after crime
                winPercentageAfterArrest = winsAfterArrest / gamesAfterArrest
                #win% before arrest must not be 0 for calculation
                if winPercentageB4Arrest > .0001: 
                    #(V2−V1)|V1|×100= win%change
                    winPercentageChange = (winPercentageAfterArrest - winPercentageB4Arrest) / winPercentageB4Arrest * 100
                    #return the win% change
                    return(round(winPercentageChange,2))

#Takeaways for Team Win% Before and After Crime Function

#The average win% change is +2.69% after players commit crimes
#If we were to take this literally, NFL players should all start committing crimes to boost their win%, but this is obviously not legitimate
#As I was creating this function, it became more and more obvious that the results would be skewed for a few reasons:
    #some of the win% changes were huge, like +300% win rate boosts. Take Kyle Lauletta for example who was arrested on 2018-10-30 while playing for the NY Giants
    #up until Lauletta's arrest, the Giants were 1-7
    #after Lauletta's arrest, the giants won 4 games and lost 4 games to finish the season 5-11, which comes out to more than a 300% percent increase in win%   
    #There were other cases where a player would be arrested after losing the first game of the season, then the team would go on to have a good season, causing massive percentage increases 

#It's clear that this function can't really be taken seriously as an indicator that crimes committed affects win% in any capacity
